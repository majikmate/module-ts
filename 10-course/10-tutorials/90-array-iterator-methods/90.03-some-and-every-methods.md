# Some and Every Methods

```typescript
// some() returns true if at least one element matches
// every() returns true if all elements match

const numbers = [1, 2, 3, 4, 5];

// Check if some numbers are even
const hasEven = numbers.some((num) => num % 2 === 0);
console.log("Has even numbers:", hasEven); // true

// Check if all numbers are positive
const allPositive = numbers.every((num) => num > 0);
console.log("All positive:", allPositive); // true

// Check if all numbers are even
const allEven = numbers.every((num) => num % 2 === 0);
console.log("All even:", allEven); // false

// some() with strings
const names = ["Alice", "Bob", "Charlie"];
const hasLongName = names.some((name) => name.length > 6);
console.log("Has long name:", hasLongName); // true

// every() with strings
const allShortNames = names.every((name) => name.length < 10);
console.log("All names short:", allShortNames); // true

// Practical example: validation
const ages = [18, 21, 25, 30];
const allAdults = ages.every((age) => age >= 18);
console.log("All adults:", allAdults); // true

const hasMinor = ages.some((age) => age < 18);
console.log("Has minor:", hasMinor); // false
```
