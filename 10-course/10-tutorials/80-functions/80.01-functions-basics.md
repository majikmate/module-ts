# Functions Basics

```typescript
// Define a function that greets a person
function greet(name: string): void {
    console.log("Hello, " + name + "!");
}

// Call the function
greet("Alice");

// Define a function that returns a value
function add(a: number, b: number): number {
    return a + b;
}

// Use the returned value
const sum = add(5, 3);
console.log("Sum:", sum);
```
