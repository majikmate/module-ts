# Arrow Functions

```typescript
// Arrow functions provide a shorter syntax for function expressions

// Traditional function expression
const addTraditional = function (a: number, b: number): number {
    return a + b;
};

// Arrow function - shorter syntax
const addArrow = (a: number, b: number): number => {
    return a + b;
};

// Arrow function with implicit return (for single expressions)
const addShort = (a: number, b: number): number => a + b;

console.log(addTraditional(2, 3));
console.log(addArrow(2, 3));
console.log(addShort(2, 3));

// Arrow function with single parameter (parentheses optional)
const square = (x: number): number => x * x;
console.log("Square of 5:", square(5));

// Arrow function with no parameters
const greet = (): void => {
    console.log("Hello, World!");
};
greet();
```
